    <script>
        // Outer space canvas with white particles + venom web click effect
        // FIXED: Small white particles now float VERY slowly and smoothly at all times
        (function(){
            const canvas = document.getElementById('particles-canvas');
            const ctx = canvas.getContext('2d');

            let W, H, particles, clickBursts = [];
            const COUNT = 160;
            const MAX_DIST = 130;

            function resize() {
                W = canvas.width = window.innerWidth;
                H = canvas.height = window.innerHeight;
            }

            function rand(min, max) {
                return Math.random() * (max - min) + min;
            }

            function createParticles() {
                particles = Array.from({ length: COUNT }, () => ({
                    x: rand(0, W),
                    y: rand(0, H),
                    vx: rand(-0.035, 0.035),
                    vy: rand(-0.035, 0.035),
                    r: rand(0.8, 2.2)
                }));
            }

            function createBurst(x, y) {
                clickBursts.push({
                    x: x,
                    y: y,
                    life: 85,
                    maxLife: 85,
                    seed: Math.random() * Math.PI * 2
                });
            }

            function draw() {
                ctx.clearRect(0, 0, W, H);

                // Subtle white connections (space web)
                for (let i = 0; i < COUNT; i++) {
                    for (let j = i + 1; j < COUNT; j++) {
                        const dx = particles[i].x - particles[j].x;
                        const dy = particles[i].y - particles[j].y;
                        const dist = Math.sqrt(dx * dx + dy * dy);

                        if (dist < MAX_DIST) {
                            ctx.beginPath();
                            ctx.strokeStyle = `rgba(255,255,255,${0.07 * (1 - dist / MAX_DIST)})`;
                            ctx.lineWidth = 0.7;
                            ctx.moveTo(particles[i].x, particles[i].y);
                            ctx.lineTo(particles[j].x, particles[j].y);
                            ctx.stroke();
                        }
                    }
                }

                // Small white particles (now slow floating)
                particles.forEach(p => {
                    ctx.beginPath();
                    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
                    ctx.fillStyle = '#ffffff';
                    ctx.fill();
                });

                // Venom web click bursts
                clickBursts.forEach(b => {
                    const progress = b.life / b.maxLife;
                    const length = 280 * (1 - progress);
                    const opacity = progress * 0.75;

                    ctx.save();
                    ctx.globalAlpha = opacity;
                    ctx.strokeStyle = '#ffffff';
                    ctx.lineWidth = 1.8;

                    const strands = 16;
                    for (let i = 0; i < strands; i++) {
                        const angle = (i * Math.PI * 2 / strands) + b.seed;
                        const ex = b.x + Math.cos(angle) * length;
                        const ey = b.y + Math.sin(angle) * length;

                        ctx.beginPath();
                        ctx.moveTo(b.x, b.y);
                        ctx.lineTo(ex, ey);
                        ctx.stroke();
                    }
                    ctx.restore();
                });
            }

            function update() {
                particles.forEach(p => {
                    p.x += p.vx;
                    p.y += p.vy;

                    if (p.x < 0) p.x = W;
                    if (p.x > W) p.x = 0;
                    if (p.y < 0) p.y = H;
                    if (p.y > H) p.y = 0;
                });

                clickBursts = clickBursts.filter(b => {
                    b.life--;
                    return b.life > 0;
                });
            }

            function loop() {
                update();
                draw();
                requestAnimationFrame(loop);
            }

            function handleClick(e) {
                const rect = canvas.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                createBurst(x, y);
            }

            function handleTouch(e) {
                if (e.changedTouches && e.changedTouches.length > 0) {
                    const rect = canvas.getBoundingClientRect();
                    const touch = e.changedTouches[0];
                    const x = touch.clientX - rect.left;
                    const y = touch.clientY - rect.top;
                    createBurst(x, y);
                }
            }

            window.addEventListener('resize', () => {
                resize();
                createParticles();
            });

            document.addEventListener('click', handleClick);
            document.addEventListener('touchend', handleTouch);

            resize();
            createParticles();
            loop();
        })();
    </script>
